`"use client"

import { useState } from "react"
import { Button } from "@/components/ui/button"
import { Card, CardContent } from "@/components/ui/card"
import { Progress } from "@/components/ui/progress"
import { CheckCircle2, XCircle, RotateCcw, Code2 } from "lucide-react"
import { questions } from "@/lib/questions"

export default function JavaMCQPractice() {
  const [currentQuestion, setCurrentQuestion] = useState(0)
  const [selectedAnswers, setSelectedAnswers] = useState<Record<number, string[]>>({})
  const [showResults, setShowResults] = useState(false)
  const [score, setScore] = useState(0)
  const [answeredQuestions, setAnsweredQuestions] = useState<Set<number>>(new Set())

  const question = questions[currentQuestion]
  const progress = ((currentQuestion + 1) / questions.length) * 100
  const isAnswered = answeredQuestions.has(currentQuestion)
  const currentAnswer = selectedAnswers[currentQuestion] || []

  const handleAnswerSelect = (option: string) => {
    if (isAnswered) return

    const newAnswers = { ...selectedAnswers }
    if (question.type === "multiple") {
      const current = newAnswers[currentQuestion] || []
      if (current.includes(option)) {
        newAnswers[currentQuestion] = current.filter((a) => a !== option)
      } else {
        newAnswers[currentQuestion] = [...current, option]
      }
    } else {
      newAnswers[currentQuestion] = [option]
    }
    setSelectedAnswers(newAnswers)
  }

  const handleSubmitAnswer = () => {
    if (currentAnswer.length === 0) return

    const correctAnswers = question.answer.split(",").map((a) => a.trim())
    const userAnswers = currentAnswer.sort()
    const isCorrect = JSON.stringify(userAnswers) === JSON.stringify(correctAnswers.sort())

    if (isCorrect) {
      setScore(score + 1)
    }

    setAnsweredQuestions(new Set([...answeredQuestions, currentQuestion]))
  }

  const handleNext = () => {
    if (currentQuestion < questions.length - 1) {
      setCurrentQuestion(currentQuestion + 1)
    } else {
      setShowResults(true)
    }
  }

  const handlePrevious = () => {
    if (currentQuestion > 0) {
      setCurrentQuestion(currentQuestion - 1)
    }
  }

  const handleRestart = () => {
    setCurrentQuestion(0)
    setSelectedAnswers({})
    setShowResults(false)
    setScore(0)
    setAnsweredQuestions(new Set())
  }

  const getAnswerStatus = (option: string) => {
    if (!isAnswered) return null

    const correctAnswers = question.answer.split(",").map((a) => a.trim())
    const isCorrectOption = correctAnswers.includes(option)
    const isSelected = currentAnswer.includes(option)

    if (isCorrectOption && isSelected) return "correct"
    if (isCorrectOption && !isSelected) return "missed"
    if (!isCorrectOption && isSelected) return "wrong"
    return null
  }

  if (showResults) {
    return (
      <div className="min-h-screen bg-background flex items-center justify-center p-4">
        <Card className="w-full max-w-2xl border-2">
          <CardContent className="p-8 text-center space-y-6">
            <div className="w-24 h-24 rounded-full bg-primary/10 mx-auto flex items-center justify-center">
              <Code2 className="w-12 h-12 text-primary" />
            </div>
            <div>
              <h1 className="text-4xl font-bold mb-2">Quiz Complete!</h1>
              <p className="text-muted-foreground text-lg">You've finished all {questions.length} Java MCQ questions</p>
            </div>
            <div className="bg-muted rounded-lg p-6">
              <div className="text-6xl font-bold mb-2">
                {score}/{questions.length}
              </div>
              <div className="text-xl text-muted-foreground">
                {Math.round((score / questions.length) * 100)}% Correct
              </div>
            </div>
            <Button onClick={handleRestart} size="lg" className="gap-2">
              <RotateCcw className="w-5 h-5" />
              Start Over
            </Button>
          </CardContent>
        </Card>
      </div>
    )
  }

  return (
    <div className="min-h-screen bg-background">
      {/* Header */}
      <header className="border-b bg-card">
        <div className="container max-w-5xl mx-auto px-4 py-4">
          <div className="flex items-center justify-between">
            <div className="flex items-center gap-3">
              <div className="w-10 h-10 rounded-lg bg-primary flex items-center justify-center">
                <Code2 className="w-6 h-6 text-primary-foreground" />
              </div>
              <div>
                <h1 className="text-xl font-bold">Java MCQ Practice</h1>
                <p className="text-sm text-muted-foreground">Master your Java skills</p>
              </div>
            </div>
            <div className="text-right">
              <div className="text-2xl font-bold">
                {score}/{answeredQuestions.size}
              </div>
              <div className="text-sm text-muted-foreground">Correct</div>
            </div>
          </div>
        </div>
      </header>

      {/* Progress Bar */}
      <div className="border-b bg-card">
        <div className="container max-w-5xl mx-auto px-4 py-3">
          <div className="flex items-center gap-4">
            <span className="text-sm font-medium whitespace-nowrap">
              Question {currentQuestion + 1} of {questions.length}
            </span>
            <Progress value={progress} className="flex-1" />
            <span className="text-sm text-muted-foreground whitespace-nowrap">{Math.round(progress)}%</span>
          </div>
        </div>
      </div>

      {/* Question Content */}
      <main className="container max-w-5xl mx-auto px-4 py-8">
        <Card className="border-2">
          <CardContent className="p-8 space-y-6">
            {/* Question */}
            <div>
              <div className="flex items-start gap-3 mb-4">
                <div className="w-8 h-8 rounded-full bg-primary/10 flex items-center justify-center shrink-0 mt-1">
                  <span className="text-sm font-bold text-primary">Q</span>
                </div>
                <div className="flex-1">
                  <h2 className="text-xl font-semibold leading-relaxed text-balance">{question.question}</h2>
                </div>
              </div>

              {/* Code Block if present */}
              {question.code && (
                <div className="bg-muted rounded-lg p-4 font-mono text-sm overflow-x-auto border">
                  <pre className="text-foreground">{question.code}</pre>
                </div>
              )}
            </div>

            {/* Options */}
            {question.options && (
              <div className="space-y-3">
                {question.options.map((option) => {
                  const status = getAnswerStatus(option.value)
                  const isSelected = currentAnswer.includes(option.value)

                  return (
                    <button
                      key={option.value}
                      onClick={() => handleAnswerSelect(option.value)}
                      disabled={isAnswered}
                      className={`w-full text-left p-4 rounded-lg border-2 transition-all ${
                        status === "correct"
                          ? "bg-green-500/10 border-green-500 text-green-600 dark:text-green-400"
                          : status === "wrong"
                            ? "bg-red-500/10 border-red-500 text-red-600 dark:text-red-400"
                            : status === "missed"
                              ? "bg-yellow-500/10 border-yellow-500 text-yellow-600 dark:text-yellow-400"
                              : isSelected
                                ? "bg-primary/10 border-primary"
                                : "bg-card border-border hover:border-primary/50"
                      } ${isAnswered ? "cursor-default" : "cursor-pointer"}`}
                    >
                      <div className="flex items-start gap-3">
                        <div
                          className={`w-6 h-6 rounded-full border-2 flex items-center justify-center shrink-0 mt-0.5 ${
                            status === "correct"
                              ? "bg-green-500 border-green-500"
                              : status === "wrong"
                                ? "bg-red-500 border-red-500"
                                : status === "missed"
                                  ? "bg-yellow-500 border-yellow-500"
                                  : isSelected
                                    ? "bg-primary border-primary"
                                    : "border-border"
                          }`}
                        >
                          {status === "correct" && <CheckCircle2 className="w-4 h-4 text-white" />}
                          {status === "wrong" && <XCircle className="w-4 h-4 text-white" />}
                          {status === "missed" && <span className="text-xs text-white">!</span>}
                          {!status && isSelected && <div className="w-2 h-2 rounded-full bg-primary-foreground" />}
                        </div>
                        <div className="flex-1">
                          <span className="font-semibold mr-2">{option.value}.</span>
                          <span>{option.text}</span>
                        </div>
                      </div>
                    </button>
                  )
                })}
              </div>
            )}

            {/* Show answer for text-based questions */}
            {!question.options && isAnswered && (
              <div className="bg-muted rounded-lg p-4 border-2 border-primary">
                <div className="font-semibold mb-1">Correct Answer:</div>
                <div className="text-primary font-mono">{question.answer}</div>
              </div>
            )}

            {/* Action Buttons */}
            <div className="flex items-center justify-between pt-4 border-t">
              <Button variant="outline" onClick={handlePrevious} disabled={currentQuestion === 0}>
                Previous
              </Button>

              <div className="flex gap-3">
                {!isAnswered && question.options && (
                  <Button onClick={handleSubmitAnswer} disabled={currentAnswer.length === 0} className="gap-2">
                    <CheckCircle2 className="w-4 h-4" />
                    Submit Answer
                  </Button>
                )}

                {(isAnswered || !question.options) && (
                  <Button onClick={handleNext} className="gap-2">
                    {currentQuestion === questions.length - 1 ? "Finish Quiz" : "Next Question"}
                  </Button>
                )}
              </div>
            </div>

            {/* Hint for multiple answers */}
            {question.type === "multiple" && !isAnswered && (
              <div className="text-sm text-muted-foreground text-center pt-2">
                ðŸ’¡ This question has multiple correct answers
              </div>
            )}
          </CardContent>
        </Card>
      </main>
    </div>
  )
}
`

`"use client"

import * as React from "react"
import * as ProgressPrimitive from "@radix-ui/react-progress"

import { cn } from "@/lib/utils"

const Progress = React.forwardRef<
  React.ElementRef<typeof ProgressPrimitive.Root>,
  React.ComponentPropsWithoutRef<typeof ProgressPrimitive.Root>
>(({ className, value, ...props }, ref) => (
  <ProgressPrimitive.Root
    ref={ref}
    className={cn("relative h-2 w-full overflow-hidden rounded-full bg-secondary", className)}
    {...props}
  >
    <ProgressPrimitive.Indicator
      className="h-full w-full flex-1 bg-primary transition-all"
      style={{ transform: `translateX(-${100 - (value || 0)}%)` }}
    />
  </ProgressPrimitive.Root>
))
Progress.displayName = ProgressPrimitive.Root.displayName

export { Progress }
`
